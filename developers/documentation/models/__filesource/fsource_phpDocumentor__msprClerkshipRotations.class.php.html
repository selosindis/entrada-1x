  7.     /**
  8.      * 
  9.      * @param mixed $rotations can be an array or a ClerkshipRotations Collection
  10.      * @param string $distance e.g. "+1 week" format sensitive.
  11.      */
  12.     public static function merge_clerkship_rotations($rotations, $distance) {
  13.         if ($rotations instanceof Collection) {
  14.             $original = $rotations;
  15.             $rotations = $rotations->container;
  16.             $revert = true;
  17.             
  18.         }
  19.         if ($rotations && count($rotations) > 1) {
  20.             for ($i = 0; $i < count($rotations) - 1; $i++) { //not caching the length as it may change, doing -1 in the condition as we're comparing consecutive elements
  21.                 $element = $rotations[$i];
  22.                 $next_element = $rotations[$i+1];
  23.                 if ($element->getTitle() == $next_element->getTitle()) { //only continue if the titles are the same
  24.                     $cur_end = $element->getFinish();
  25.                     $next_start = $element->getStart();
  26.                     $cur_end_mod = strtotime(date(DATE_RFC2822, $cur_end) . " " . $distance);
  27.                     if ($cur_end_mod >= $next_start) { //overlapping or meeting
  28.                         //merge
  29.                         $title = $element->getTitle();
  30.                         $start = $element->getStart();
  31.                         $finish = $next_element->getFinish();
  32.                         $user = $element->getUser();
  33.                         $user_id = $user->getID(); 
  34.                         $new_element = new ClerkshipRotation($user_id, $title, $start, $finish, true);
  35.                         array_splice($rotations, $i, 2, array($new_element));
  36.                         $i--;
  37.                     }
  38.                 } 
  39.             }
  40.         }
  41.         
  42.         if ($revert) {
  43.             $original->container = $rotations;
  44.         }
  45.         
  46.         return $rotations;
  47.     }
